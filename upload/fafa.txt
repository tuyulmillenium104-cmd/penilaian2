// TOKEN INSTANCE MANAGER
// ============================================================

class TokenInstanceManager {
  constructor() {
    this.instances = [];
  }
  
  async initialize() {
    console.log('[TokenManager] Initializing...');
    
    for (const tokenConfig of CONFIG.tokens) {
      this.instances.push({
        name: tokenConfig.name,
        config: tokenConfig.config,
        priority: tokenConfig.priority,
        sdk: null,
        remainingDaily: CONFIG.rateLimit.dailyPerToken,
        requestCount: 0,
        errorCount: 0,
        isExhausted: false
      });
      console.log(`[TokenManager] Prepared: ${tokenConfig.name}`);
    }
  }
  
  async getSDKInstance(instance) {
    // Return existing instance if already created
    if (instance.sdk) return instance.sdk;
    
    // SDK reads config from: cwd/.z-ai-config, ~/.z-ai-config, /etc/.z-ai-config
    // We'll write to current directory (has write permission)
    const configPath = path.join(process.cwd(), '.z-ai-config');
    const backupPath = path.join(process.cwd(), '.z-ai-config-backup');
    
    // Backup existing config
    try {
      if (fs.existsSync(configPath)) {
        fs.copyFileSync(configPath, backupPath);
      }
    } catch (e) {}
    
    // Write new config
    fs.writeFileSync(configPath, JSON.stringify(instance.config));
    
    try {
      // Create SDK instance
      const zai = await ZAI_SDK.default.create();
      instance.sdk = zai;
      return zai;
    } finally {
      // Restore backup (so we don't interfere with other operations)
      try {
        if (fs.existsSync(backupPath)) {
          fs.copyFileSync(backupPath, configPath);
          fs.unlinkSync(backupPath);
        }
      } catch (e) {}
    }
  }
  
  getBestInstance() {
    // Get available (non-exhausted) instances
    const available = this.instances.filter(i => !i.isExhausted);
    
    // If all exhausted, reset all
    if (available.length === 0) {
      this.instances.forEach(i => i.isExhausted = false);
      console.log('[TokenManager] All instances exhausted, resetting...');
    }
    
    // Return instance with most remaining quota
    return this.instances
      .filter(i => !i.isExhausted)
      .sort((a, b) => b.remainingDaily - a.remainingDaily)[0];
  }
  
  updateStats(name, success, error = null) {
    const instance = this.instances.find(i => i.name === name);
    if (!instance) return;
    
    if (success) {
      instance.requestCount++;
      instance.remainingDaily = Math.max(0, instance.remainingDaily - 1);
      
      // Mark exhausted if low quota
      if (instance.remainingDaily < 5) {
        instance.isExhausted = true;
      }
    } else {
      instance.errorCount++;
      
      // Mark exhausted if rate limited
      if (error?.message?.includes('429')) {
        instance.isExhausted = true;
      }
    }
  }
  
  getStatus() {
    return {
      instances: this.instances.map(i => ({
        name: i.name,
        remainingDaily: i.remainingDaily,
        requestCount: i.requestCount,
        errorCount: i.errorCount,
        isExhausted: i.isExhausted
      })),
      totalRemaining: this.instances.reduce((sum, i) => sum + i.remainingDaily, 0)
    };
  }
}

// RATE LIMITER
// ============================================================

class RateLimiter {
  constructor() {
    this.requestTimes = [];
  }
  
  async waitForSlot() {
    const now = Date.now();
    
    // Clean old requests (> 1 second ago)
    this.requestTimes = this.requestTimes.filter(t => now - t < 1000);
    
    // QPS limit is 2, wait if needed
    if (this.requestTimes.length >= CONFIG.rateLimit.qps) {
      const wait = 1000 - (now - Math.min(...this.requestTimes)) + 50;
      if (wait > 0) {
        await new Promise(r => setTimeout(r, wait));
      }
    }
    
    this.requestTimes.push(Date.now());
  }
}

// RESILIENT Z-AI SDK CLIENT
// ============================================================

class ResilientZAISDKClient {
  constructor() {
    this.tokenManager = new TokenInstanceManager();
    this.rateLimiter = new RateLimiter();
    this.ipRotator = new IPRotator();
    this.initialized = false;
  }
  
  async initialize() {
    if (this.initialized) return;
    await this.tokenManager.initialize();
    this.initialized = true;
  }
  
  /**
   * Chat completion
   * @param {Array} messages - Array of {role, content}
   * @param {Object} options - Optional: model, maxTokens, temperature
   */
  async chat(messages, options = {}) {
    if (!this.initialized) await this.initialize();
    
    let lastError = null;
    
    for (let attempt = 0; attempt < CONFIG.retry.maxRetries; attempt++) {
      try {
        // IP rotation
        await this.ipRotator.rotate();
        
        // Rate limiter
        await this.rateLimiter.waitForSlot();
        
        // Get best token instance
        const instance = this.tokenManager.getBestInstance();
        console.log(`[Client] Using ${instance.name} (attempt ${attempt + 1})`);
        
        // Get SDK instance
        const sdk = await this.tokenManager.getSDKInstance(instance);
        
        // Make request
        const result = await sdk.chat.completions.create({
          messages,
          model: options.model,
          max_tokens: options.maxTokens || 100,
          thinking: options.thinking || { type: 'disabled' }
        });
        
        // Update stats
        this.tokenManager.updateStats(instance.name, true);
        
        // Return content
        return result.choices[0]?.message?.content || result;
        
      } catch (error) {
        lastError = error;
        console.error(`[Client] Attempt ${attempt + 1} failed:`, error.message);
        
        // Update error stats
        try {
          const instance = this.tokenManager.getBestInstance();
          this.tokenManager.updateStats(instance.name, false, error);
        } catch (e) {}
        
        // Exponential backoff
        const delay = Math.min(
          CONFIG.retry.baseDelay * Math.pow(2, attempt),
          CONFIG.retry.maxDelay
        );
        await new Promise(r => setTimeout(r, delay));
      }
    }
    
    throw lastError;
  }
  
  /**
   * Web search
   * @param {string} query - Search query
   * @param {number} num - Number of results (default: 10)
   */
  async webSearch(query, num = 10) {
    if (!this.initialized) await this.initialize();
    
    await this.rateLimiter.waitForSlot();
    
    const instance = this.tokenManager.getBestInstance();
    console.log(`[Client] Web search using ${instance.name}`);
    
    const sdk = await this.tokenManager.getSDKInstance(instance);
    
    const result = await sdk.functions.invoke('web_search', { query, num });
    this.tokenManager.updateStats(instance.name, true);
    
    return result;
  }
  
  /**
   * Get client status
   */
  getStatus() {
    return {
      tokens: this.tokenManager.getStatus(),
      ipRotation: this.ipRotator.getStats(),
      initialized: this.initialized
    };
  }
}

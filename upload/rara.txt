// IP ROTATOR - Mengelola rotasi IP via NAT Pool
// ============================================================

class IPRotator {
  constructor() {
    this.enabled = CONFIG.ipRotation.enabled;
    this.currentIP = null;
    this.ipHistory = [];
    this.requestCount = 0;
    this.ipPool = [];
    
    if (this.enabled) {
      this.initialize();
    }
  }
  
  async initialize() {
    try {
      // Get current IP
      this.currentIP = await this.getCurrentIP();
      console.log(`[IPRotator] Initialized with IP: ${this.currentIP}`);
      
      // Discover NAT pool
      await this.discoverNATPool();
    } catch (error) {
      console.warn('[IPRotator] Initialization failed:', error.message);
      console.warn('[IPRotator] Continuing without IP rotation');
      this.enabled = false;
    }
  }
  
  // Get current public IP
  async getCurrentIP() {
    const services = [
      'https://api.ipify.org?format=text',
      'https://icanhazip.com',
      'https://ifconfig.me/ip',
      'https://ipecho.net/plain'
    ];
    
    for (const service of services) {
      try {
        const result = await this.httpGet(service);
        return result.trim();
      } catch (e) {
        continue;
      }
    }
    
    // Fallback: get local IP
    return this.getLocalIP();
  }
  
  // Get local IP address
  getLocalIP() {
    const interfaces = require('os').networkInterfaces();
    for (const name of Object.keys(interfaces)) {
      for (const iface of interfaces[name]) {
        if (iface.family === 'IPv4' && !iface.internal) {
          return iface.address;
        }
      }
    }
    return '127.0.0.1';
  }
  
  // HTTP GET helper
  httpGet(url) {
    return new Promise((resolve, reject) => {
      const client = url.startsWith('https') ? https : http;
      const timeout = 5000;
      
      const req = client.get(url, { timeout }, (res) => {
        let data = '';
        res.on('data', chunk => data += chunk);
        res.on('end', () => resolve(data));
      });
      
      req.on('error', reject);
      req.on('timeout', () => {
        req.destroy();
        reject(new Error('Timeout'));
      });
    });
  }
  
  // Discover NAT pool
  async discoverNATPool() {
    try {
      // Check if NAT pool is available via iptables
      const result = execSync('iptables -t nat -L POSTROUTING -n 2>/dev/null | grep SNAT || echo "no-nat"', 
        { encoding: 'utf8' });
      
      if (result.includes('no-nat')) {
        console.log('[IPRotator] No NAT pool found, using single IP mode');
        return;
      }
      
      // Parse NAT pool
      const lines = result.split('\n').filter(l => l.includes('SNAT'));
      this.ipPool = lines.map(line => {
        const match = line.match(/to:([0-9.]+)/);
        return match ? match[1] : null;
      }).filter(ip => ip);
      
      if (this.ipPool.length > 0) {
        console.log(`[IPRotator] NAT pool discovered: ${this.ipPool.length} IPs`);
      }
    } catch (error) {
      // NAT not available, will use IP rotation via interface reset
      console.log('[IPRotator] NAT pool not available, using interface reset method');
    }
  }
  
  // Rotate IP - Main method
  async rotate() {
    if (!this.enabled) {
      return this.currentIP;
    }
    
    const previousIP = this.currentIP;
    
    try {
      // Method 1: NAT Pool rotation (if available)
      if (this.ipPool.length > 0) {
        await this.rotateViaNAT();
      } 
      // Method 2: Interface reset
      else {
        await this.rotateViaReset();
      }
      
      // Verify new IP
      if (CONFIG.ipRotation.verifyRotation) {
        const newIP = await this.getCurrentIP();
        
        if (newIP !== previousIP) {
          this.ipHistory.push({
            from: previousIP,
            to: newIP,
            timestamp: Date.now()
          });
          this.currentIP = newIP;
          console.log(`[IPRotator] Rotated: ${previousIP} → ${newIP}`);
        } else {
          console.log(`[IPRotator] IP unchanged: ${newIP}`);
        }
      }
      
    } catch (error) {
      console.warn('[IPRotator] Rotation failed:', error.message);
    }
    
    this.requestCount++;
    return this.currentIP;
  }
  
  // Rotate via NAT pool
  async rotateViaNAT() {
    const mode = CONFIG.ipRotation.rotationMode;
    let nextIP;
    
    switch (mode) {
      case 'random':
        nextIP = this.ipPool[Math.floor(Math.random() * this.ipPool.length)];
        break;
      case 'least_used':
        // Track usage and pick least used
        nextIP = this.ipPool[0]; // Simplified
        break;
      case 'round_robin':
      default:
        const idx = this.requestCount % this.ipPool.length;
        nextIP = this.ipPool[idx];
    }
    
    // Set SNAT rule
    try {
      execSync(`iptables -t nat -I POSTROUTING 1 -j SNAT --to-source ${nextIP} 2>/dev/null || true`);
    } catch (e) {
      // Ignore iptables errors
    }
  }
  
  // Rotate via connection reset
  async rotateViaReset() {
    return new Promise((resolve) => {
      // Force new connection by closing existing ones
      // This typically triggers NAT to assign new IP
      
      // Method: HTTP request with connection close
      const options = {
        hostname: '172.25.136.210',
        port: 8080,
        path: '/v1/health',
        method: 'GET',
        timeout: 1000
      };
      
      const req = http.request(options, () => {
        resolve();
      });
      
      req.on('error', () => resolve());
      req.on('timeout', () => {
        req.destroy();
        resolve();
      });
      
      req.end();
    });
  }
  
  // Get stats
  getStats() {
    return {
      enabled: this.enabled,
      currentIP: this.currentIP,
      rotationCount: this.ipHistory.length,
      poolSize: this.ipPool.length,
      requestCount: this.requestCount
    };
  }
}

// TOKEN MANAGER - Mengelola rotasi token
// ============================================================

class TokenManager {
  constructor() {
    this.tokens = CONFIG.tokens.map(t => ({
      ...t,
      remainingDaily: CONFIG.rateLimit.dailyPerToken,
      remainingUserDaily: CONFIG.rateLimit.userDailyPerToken,
      lastUsed: 0,
      requestCount: 0,
      isExhausted: false
    }));
    this.currentIndex = 0;
  }
  
  // Ambil token terbaik (most remaining quota)
  getBestToken() {
    // Sort by remaining quota, priority
    const availableTokens = this.tokens.filter(t => !t.isExhausted);
    
    if (availableTokens.length === 0) {
      // All tokens exhausted - try to reset exhausted flags
      this.tokens.forEach(t => t.isExhausted = false);
      console.log('[TokenManager] All tokens were exhausted, resetting...');
    }
    
    // Get token with highest remaining quota
    const bestToken = this.tokens
      .filter(t => !t.isExhausted)
      .sort((a, b) => {
        // Prioritize by remaining daily quota
        if (b.remainingDaily !== a.remainingDaily) {
          return b.remainingDaily - a.remainingDaily;
        }
        return a.priority - b.priority;
      })[0];
    
    if (!bestToken) {
      throw new Error('No available tokens');
    }
    
    return bestToken;
  }
  
  // Update token stats dari response headers
  updateTokenStats(tokenName, headers) {
    const token = this.tokens.find(t => t.name === tokenName);
    if (!token) return;
    
    const dailyRemaining = parseInt(headers['x-ratelimit-remaining-daily'] || '0');
    const userDailyRemaining = parseInt(headers['x-ratelimit-user-daily-remaining'] || '0');
    
    token.remainingDaily = dailyRemaining;
    token.remainingUserDaily = userDailyRemaining;
    token.requestCount++;
    token.lastUsed = Date.now();
    
    // Mark as exhausted if quota low
    if (dailyRemaining < 5) {
      token.isExhausted = true;
      console.log(`[TokenManager] Token ${tokenName} nearly exhausted (${dailyRemaining} remaining)`);
    }
    
    console.log(`[TokenManager] ${tokenName}: Daily=${dailyRemaining}, UserDaily=${userDailyRemaining}`);
  }
  
  // Get status semua token
  getStatus() {
    return {
      tokens: this.tokens.map(t => ({
        name: t.name,
        remainingDaily: t.remainingDaily,
        remainingUserDaily: t.remainingUserDaily,
        isExhausted: t.isExhausted,
        requestCount: t.requestCount
      })),
      totalRemaining: this.tokens.reduce((sum, t) => sum + t.remainingDaily, 0)
    };
  }
}

// RATE LIMITER - Mengontrol request rate
// ============================================================

class RateLimiter {
  constructor() {
    this.requestTimes = [];
    this.minInterval = 1000 / CONFIG.rateLimit.qps; // 500ms untuk 2 QPS
  }
  
  async waitForSlot() {
    const now = Date.now();
    
    // Clean old requests (> 1 second ago)
    this.requestTimes = this.requestTimes.filter(t => now - t < 1000);
    
    // Check if we need to wait
    if (this.requestTimes.length >= CONFIG.rateLimit.qps) {
      const oldestRequest = Math.min(...this.requestTimes);
      const waitTime = 1000 - (now - oldestRequest) + 50; // 50ms buffer
      
      if (waitTime > 0) {
        console.log(`[RateLimiter] Waiting ${waitTime}ms for QPS limit`);
        await this.sleep(waitTime);
      }
    }
    
    this.requestTimes.push(Date.now());
  }
  
  sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}

// RESILIENT Z-AI CLIENT - Main client dengan IP + Token Rotation
// ============================================================

class ResilientZAIClient {
  constructor(options = {}) {
    this.tokenManager = new TokenManager();
    this.rateLimiter = new RateLimiter();
    this.ipRotator = new IPRotator();
    this.gatewayIndex = 0;
    this.options = options;
  }
  
  // Get next gateway (round-robin)
  getGateway() {
    const gateway = CONFIG.gateways[this.gatewayIndex];
    this.gatewayIndex = (this.gatewayIndex + 1) % CONFIG.gateways.length;
    return gateway;
  }
  
  // Main chat method
  async chat(messages, options = {}) {
    let lastError = null;
    
    for (let attempt = 0; attempt < CONFIG.retry.maxRetries; attempt++) {
      try {
        // Rotate IP before request (if enabled)
        if (this.ipRotator.enabled) {
          await this.ipRotator.rotate();
        }
        
        // Wait for rate limiter
        await this.rateLimiter.waitForSlot();
        
        // Get best token
        const token = this.tokenManager.getBestToken();
        const gateway = this.getGateway();
        
        console.log(`[Client] Attempt ${attempt + 1}: Using ${token.name} on ${gateway}`);
        if (this.ipRotator.enabled) {
          console.log(`[Client] Current IP: ${this.ipRotator.currentIP}`);
        }
        
        // Make request
        const result = await this.makeRequest(gateway, token, messages, options);
        
        return result;
        
      } catch (error) {
        lastError = error;
        console.error(`[Client] Attempt ${attempt + 1} failed:`, error.message);
        
        // Exponential backoff
        const delay = Math.min(
          CONFIG.retry.baseDelay * Math.pow(2, attempt),
          CONFIG.retry.maxDelay
        );
        console.log(`[Client] Retrying in ${delay}ms...`);
        await this.rateLimiter.sleep(delay);
      }
    }
    
    throw lastError || new Error('All retry attempts failed');
  }
  
  // Make HTTP request
  async makeRequest(gateway, token, messages, options = {}) {
    return new Promise((resolve, reject) => {
      const url = new URL(`${gateway}/chat/completions`);
      const isHttps = url.protocol === 'https:';
      const httpModule = isHttps ? https : http;
      
      const requestBody = JSON.stringify({
        messages: messages,
        model: options.model || 'glm-4-flash',
        max_tokens: options.maxTokens || 1000,
        temperature: options.temperature || 0.7,
        thinking: options.thinking || { type: 'disabled' },
        ...options.extra
      });
      
      const headers = {
        'Content-Type': 'application/json',
        'Authorization': 'Bearer Z.ai',
        'X-Z-AI-From': 'Z',
        'X-Token': token.token,
        'X-User-Id': token.userId,
        'X-Chat-Id': token.chatId
      };
      
      const reqOptions = {
        hostname: url.hostname,
        port: url.port || (isHttps ? 443 : 80),
        path: url.pathname,
        method: 'POST',
        headers: headers
      };
      
      const req = httpModule.request(reqOptions, (res) => {
        let data = '';
        
        res.on('data', chunk => {
          data += chunk;
        });
        
        res.on('end', () => {
          // Update token stats from headers
          this.tokenManager.updateTokenStats(token.name, res.headers);
          
          if (res.statusCode === 200) {
            try {
              const json = JSON.parse(data);
              resolve(json);
            } catch (e) {
              reject(new Error(`JSON parse error: ${e.message}`));
            }
          } else if (res.statusCode === 429) {
            // Rate limited - mark token as exhausted temporarily
            const tokenObj = this.tokenManager.tokens.find(t => t.name === token.name);
            if (tokenObj) tokenObj.isExhausted = true;
            reject(new Error(`Rate limited (429): ${data}`));
          } else {
            reject(new Error(`HTTP ${res.statusCode}: ${data}`));
          }
        });
      });
      
      req.on('error', (e) => {
        reject(new Error(`Request error: ${e.message}`));
      });
      
      req.setTimeout(30000, () => {
        req.destroy();
        reject(new Error('Request timeout'));
      });
      
      req.write(requestBody);
      req.end();
    });
  }
  
  // Web search method
  async webSearch(query, num = 10) {
    // Rotate IP before request (if enabled)
    if (this.ipRotator.enabled) {
      await this.ipRotator.rotate();
    }
    
    await this.rateLimiter.waitForSlot();
    
    const token = this.tokenManager.getBestToken();
    const gateway = this.getGateway();
    
    return new Promise((resolve, reject) => {
      const url = new URL(`${gateway}/functions/invoke`);
      const isHttps = url.protocol === 'https:';
      const httpModule = isHttps ? https : http;
      
      const requestBody = JSON.stringify({
        function_name: 'web_search',
        arguments: { query, num }
      });
      
      const headers = {
        'Content-Type': 'application/json',
        'Authorization': 'Bearer Z.ai',
        'X-Z-AI-From': 'Z',
        'X-Token': token.token
      };
      
      const reqOptions = {
        hostname: url.hostname,
        port: url.port || (isHttps ? 443 : 80),
        path: url.pathname,
        method: 'POST',
        headers: headers
      };
      
      const req = httpModule.request(reqOptions, (res) => {
        let data = '';
        
        res.on('data', chunk => { data += chunk; });
        res.on('end', () => {
          this.tokenManager.updateTokenStats(token.name, res.headers);
          
          if (res.statusCode === 200) {
            try {
              const json = JSON.parse(data);
              resolve(json.result);
            } catch (e) {
              reject(new Error(`JSON parse error: ${e.message}`));
            }
          } else {
            reject(new Error(`HTTP ${res.statusCode}: ${data}`));
          }
        });
      });
      
      req.on('error', reject);
      req.setTimeout(30000, () => {
        req.destroy();
        reject(new Error('Request timeout'));
      });
      
      req.write(requestBody);
      req.end();
    });
  }
  
  // Get complete status
  getStatus() {
    return {
      tokens: this.tokenManager.getStatus(),
      ipRotation: this.ipRotator.getStats(),
      gateways: CONFIG.gateways
    };
  }
}
